Question 1

char_to_words = {}

input_file = open("input.txt", "r")

for line in input_file:
    for word in line.split():
        first_char = word[0].lower()
        if not first_char in char_to_words:
            char_to_words[first_char] = []

        char_to_words[first_char].append(word.lower())
        
input_file.close()

output_file = open("output.txt", "w")

for letter in sorted(char_to_words):
    output_file.write(letter.upper() + "\n")
    for word in sorted(char_to_words[letter]):
        output_file.write(word+"\n")
    output_file.write("\n")

output_file.close()

Question 3

def get_available_tas(ta_to_time: dict, time: str) -> set:

	available_tas = set()

	for ta in ta_to_time:
		times = ta_to_time[ta]
		if(time in times):
			available_tas.add(ta)
	
	return available_tas
			
def get_ta_least_sessions(available_tas: set) -> str:

	min_ta = ""
	min_sessions = float('inf')
	
	for ta in available_tas:
		sessions = len(ta_to_time[ta])
		if(sessions < min_sessions):
			min_ta = ta
			min_sessions = sessions
			
	return min_ta

def ta_to_assign(ta_to_time: dict, time: str) -> str:

	available_tas = get_available_tas(ta_to_time, time)
	return get_ta_least_sessions(available_tas)
